function get_keyrange
  $group_iterator := 1
  $keys := 0
  while($keys <= 127) 
    $blip_note := play_note($keys, 1, 0, 1)
    set_event_par_arr($blip_note, $EVENT_PAR_ALLOW_GROUP, 0, $ALL_GROUPS)
    change_vol($blip_note, -2000000, 0)
    set_event_par_arr($blip_note, $EVENT_PAR_ALLOW_GROUP, 1, $group_iterator)
    wait(1)
    $zone_id := get_event_par($blip_note, $EVENT_PAR_ZONE_ID)
    if(($zone_id # -1) and ($zone_id # 0))
      %range[0] := $keys
      $keys := 127
    end if
    fade_out($blip_note, 0, 1)
    inc($keys)
  end while
  $keys := 0
  while($keys <= 127) 
    $blip_note := play_note(127 - $keys, 1, 0, 1)
    set_event_par_arr($blip_note, $EVENT_PAR_ALLOW_GROUP, 0, $ALL_GROUPS)
    change_vol($blip_note, -2000000, 0)
    set_event_par_arr($blip_note, $EVENT_PAR_ALLOW_GROUP, 1, $group_iterator)
    wait(1)
    $zone_id := get_event_par($blip_note, $EVENT_PAR_ZONE_ID)
    if(($zone_id # -1) and ($zone_id # 0))
      %range[1] := 127 - $keys
      $keys := 127
    end if
    fade_out($blip_note, 0, 1)
    inc($keys)
  end while
end function

  end while

on note
  ignore_event($EVENT_ID)
  $event_note := $EVENT_NOTE
  $event_velocity := $EVENT_VELOCITY
  if(- sh_right(abs(%development[0] - 22714) - 1, 31) = 1)
    call get_event_notes
    if(in_range($event_note, %range[0] - 1, %range[1] + 1))
      sort(%event_notes, 1)
      $btn_rr_sum := 0
      $u := 0
      while($u <= (num_elements(%btn_rr) - 1)) 
        $btn_rr_sum := %btn_rr_values[$u] + $btn_rr_sum
        inc($u)
      end while
      if($btn_rr_sum # 0)
        %rr_count[$event_note] := (%rr_count[$event_note] + 1) mod ($NUM_GROUPS - 1)
        while(%btn_rr_values[%rr_count[$event_note]] # 1) 
          %rr_count[$event_note] := (%rr_count[$event_note] + 1) mod ($NUM_GROUPS - 1)
        end while
        select($INSTRUMENT_TYPE)
          case 1
            %event_ids[$event_note] := play_note($event_note, $event_velocity, 0, -1)
          case 0
            %event_ids[$event_note] := play_note($event_note, $event_velocity, 0, 0)
        end select
        set_event_par_arr(%event_ids[$event_note], $EVENT_PAR_ALLOW_GROUP, 0, $ALL_GROUPS)
        set_event_par_arr(%event_ids[$event_note], $EVENT_PAR_ALLOW_GROUP, 1, $rr_group + %rr_count[$event_note])
        if(search(%event_notes, -1) = 1)
          wait(1)
          set_control_par_str(get_ui_id($lbl_filename), $CONTROL_PAR_TEXT, "Filename: " & get_sample(get_event_par(%event_ids[$event_note], $EVENT_PAR_ZONE_ID), $NI_FILE_NAME))
          $v := 0
          while($v <= (($NUM_GROUPS - 1) - 1)) 
            set_control_par(%lbl_rr[$v], $CONTROL_PAR_FONT_TYPE, %_font__3pt3[0])
            inc($v)
          end while
          set_control_par(%lbl_rr[%rr_count[$event_note]], $CONTROL_PAR_FONT_TYPE, %_font__3pt3[2])
        else
          set_control_par_str(get_ui_id($lbl_filename), $CONTROL_PAR_TEXT, "Polyphonic")
        end if
      end if
    end if
  end if
  if(not (- sh_right(abs(%development[0] - 22714) - 1, 31) = 1))
    if(%event_notes[$event_note] # $event_note)
      %event_notes[$event_note] := $event_note
      %rr_count[$event_note] := (%rr_count[$event_note] + 1) mod $NUM_RR_FINAL
      %event_ids[$event_note] := play_note($event_note, $event_velocity, 0, ($INSTRUMENT_TYPE * -1) + abs(sh_right(abs($INSTRUMENT_TYPE) - 1, 31)))
      set_event_par_arr(%event_ids[$event_note], $EVENT_PAR_ALLOW_GROUP, 0, $ALL_GROUPS)
      set_event_par_arr(%event_ids[$event_note], $EVENT_PAR_ALLOW_GROUP, 1, %rr_count[$event_note] + $rr_group)
    end if
  end if
end on
