on init
  declare $loc_2
  declare ui_button $btn_1 
  declare ui_button $btn_2 
  pgs_create_key( AUTO_SAVE, 1)
  set_listener($NI_SIGNAL_TIMER_MS, 1000000)
  declare $autoSave
end on
on listener
  if($NI_SIGNAL_TYPE = $NI_SIGNAL_TIMER_MS)
    $autoSave := ($autoSave + 1) mod 60
    if($autoSave = 0)
      pgs_set_key_val( AUTO_SAVE, 0, 1)
    end if
  end if
end on
on ui_control($btn_1)
  message("sdlkj")
  $loc_2 := 0
  inc($loc_2)
  message(1)
  message($loc_2 & ", " & 1)
end on
on ui_control($btn_2)
  message("sdlkj")
  $loc_2 := 0
  inc($loc_2)
  message(1)
  message($loc_2 & ", " & 4)
  message(1)
end on
