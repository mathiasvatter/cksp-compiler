function floatmask.get(float_mask[1] : int[], idx : int) -> return : int
	return := sh_right(float_mask[idx / 32], (31 - idx) mod 32) .and. 1
end function

function array.map(array[1], func) -> return
	i := 0
	while(i <= (num_elements(array) - 1)) 
		array[i] := func(array[i])
		inc(i)
	end while
end function

function floatmask.print(float_mask[1] : int[], loc_5 : string)

	i := 0
	while(i <= (64 - 1)) 
		loc_5 := loc_5 & floatmask.get(float_mask, i)

		end if

	end while
	loc_5 := loc_5 & "]"
	message(loc_5)
end function
