function floatmask.get(float_mask[1] : int[], idx : int) -> return : int
	return := sh_right(float_mask[idx / 32], (31 - idx) mod 32) .and. 1
end function

function array.map(array, func) -> return
	i := 0
	while(i <= (num_elements(array) - 1)) 
		array[i] := func(array[i])
		inc(i)
	end while
end function

function floatmask.print(float_mask : int[])
	declare bl : string := "["
	i := 0
	while(i <= (64 - 1)) 
		bl := bl & floatmask.get(float_mask, i)
		if(i < (64 - 1))
			bl := bl & ", "
		end if
		inc(i)
	end while
	bl := bl & "]"
	message(bl)
end function

function array.clear(array : int[])
	i := 0
	while(i <= (num_elements(array) - 1)) 
		array[i] := 0
		inc(i)
	end while
end function
