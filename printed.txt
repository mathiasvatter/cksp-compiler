end function

function array.reduce(array : unknown, func : unknown) -> result : unknown
	declare lib.i : unknown
	result := array[0]
	lib.i := 0
	while(lib.i <= (num_elements(array) - 1)) 
		result := func(result, array[lib.i]) : unknown
		inc(lib.i)
	end while
end function
