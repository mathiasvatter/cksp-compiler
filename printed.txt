	message("hallo " & 12 & " und du " & 3 & ""){Builtin, None, message}
	set_listener(NI_SIGNAL_TIMER_MS{int} {NI_SIGNAL_TIMER_MS} , 1000){Builtin, None, set_listener}
	declare flag : int {Reuses: 0}

on note{thread-unsafe}
	flag{int} {flag}  := 1
	call do_something{UserDefined, Call, do_something}
	flag{int} {flag}  := 0
end on

on listener{thread-unsafe}
	flag{int} {flag}  := 1
	call do_something{UserDefined, Call, do_something}
	flag{int} {flag}  := 0
end on

function do_something{thread-unsafe}
	if(flag{int} {flag}  = 1)
		message("this function is already running, don't run it again"){Builtin, None, message}
		exit{Builtin, None, exit}
	else
		message("good to go"){Builtin, None, message}
		wait(DURATION_QUARTER{int} {DURATION_QUARTER} ){Builtin, None, wait}
	end if
end function
