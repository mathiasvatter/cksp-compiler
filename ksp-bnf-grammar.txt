





<array> ::= ("!" | "%" | "?")? <keyword> "[" (<num> ("," <num>)? )? "]"
<constant> ::= "const" <variable>

<array_declaration> ::= <array> ":=" "(" <int> ("," <int>)? ")"
<constant_declaration> ::= <constant> ":=" (<string> | <num>)
<declaration> ::= "declare " (<constant_declaration> | 
<variable_declaration> | <array_declaration>)
<init_callback> ::= "\n" "on init" "\n" <declaration>* "\n" "end on" "\n"


<int> ::= ("-")? [1-9]+ | "int(" <real> ")"
<real> ::= ("-")? <int> "." <int> | "real(" <int> ")"
<num> ::= (<int> | <real>)
<binary_op> ::= - | + | mod | / | * | <bitwise_op>
<linebreak> ::= \n
<assignment> ::= (<variable> | <array>) ":=" (<num> | <string>)
<comparison> ::= "=" | "<" | ">" | "<=" | ">=" | "!="
<bool> ::= and | or | not
<bitwise_op> ::= .and. | .xor. | .or. | .not.

<function_header> ::= <keyword> "(" (<expression> ("," <expression>)?)? ")"
<function_definition> ::= "function" <function_header> ("->" <variable>)? <linebreak> <statements> "end function"
<function_call> ::= "call" <function_header>


<keyword> ::= ([0-9] | [a-z] | [A-Z] | "_")+ | <keyword> "." <keyword>
<string> ::= "\'" <text> "\'"
<variable_identifier> ::= "$" | "~" | "@"
<array_identifier> ::= "!" | "%" | "?"
<variable> ::= <variable_identifier>? <keyword>
<variable_declaration> ::= "declare" <variable> | "declare" <variable_assign>
<variable_assign> ::= <variable> ":=" (<expression> | <string>) <linebreak>

<assign_statement> ::= <variable_assign> | <array_assign>
<binary_expression> ::= (<int> | <variable> | <function_header>) <binary_op> <binary_expression> | (<int> | <variable> | <function_header>)
<real_expression> ::= (<real> | <variable>
<string_expression> ::= (<string> | <variable> | <function_header>) (& (<string> | <variable> | <function_header>))?
<expression> ::= <binary_expression> | <real_expression> | <string_expression>
<statements> ::= (<if_statement> | <while_statement> | <for_statement> | <select_statement> | <assign_statement)*
<for_statement> ::= for <variable> := <expression> <linebreak> <statements> end for <linebreak>
<condition> ::= <expression> <comparison> <expression> (<bool> <condition>)*
<if_statement> ::= if (? <condition> )? <linebreak> <statements> else? <linebreak> <statements> end if <linebreak>
<while_statement> ::= while (? <condition> )? <linebreak> <statements> end while <linebreak>
<select_statement> ::= select (? <expression> )? <linebreak> (case <expression> <linebreak> <statements>)+ end select <linebreak>

<callback_identifier> ::= ("async_complete" | "controller" | "init" |
                          "listener" | "note" | "persistence_changed" | "pgs_changed" | "poly_at" |
                          "release" | "rpn/nrpn" | "ui_control" | "ui_update")
<callback> ::= "on" <callback_identifier> <linebreak> <statements> "end on"
<programm> ::= ((<callback> | <function_definition>) <linebreak>)*
